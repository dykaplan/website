  ◼  Custom footer script

  ◼  — runs AFTER the theme’s main.min.js

  ◼  — moves the links back into <ul class="visible-links">

  ◼  — removes the hamburger button

{% endcomment %}


<script>

document.addEventListener('DOMContentLoaded', function () {

  var nav     = document.querySelector('.greedy-nav');

  if (!nav)  { return; }


  /* 1 — move every <li> out of .hidden-links into .visible-links */

  var visible = nav.querySelector('.visible-links');

  var hidden  = nav.querySelector('.hidden-links');

  if (visible && hidden) {

    while (hidden.firstChild) { visible.appendChild(hidden.firstChild); }

    hidden.remove();                                 // throw the list away

  }


  /* 2 — remove the toggle button entirely */

  var toggle = nav.querySelector('.greedy-nav__toggle');

  if (toggle) { toggle.remove(); }

});

</script>
