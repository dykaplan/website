  ◼  Custom footer script

  ◼  — runs AFTER the theme’s main.min.js

  ◼  — moves the links back into <ul class="visible-links">

  ◼  — removes the hamburger button

{% endcomment %}


<script>

(function () {

  function restoreMenu () {

    var nav = document.querySelector('.greedy-nav');

    if (!nav) { return; }


    /* 1 ─ move every <li> back to the visible list */

    var vis = nav.querySelector('.visible-links');

    var hid = nav.querySelector('.hidden-links');

    if (vis && hid) {

      while (hid.firstChild) { vis.appendChild(hid.firstChild); }

    }


    /* 2 ─ kill the inline styles Greedy-Nav sets */

    if (hid) {

      hid.style.display    = 'flex';

      hid.style.opacity    = '1';

      hid.style.transform  = 'none';

    }


    /* 3 ─ hide the hamburger button */

    var toggle = nav.querySelector('.greedy-nav__toggle');

    if (toggle) { toggle.style.display = 'none'; }

  }


  /* run once after everything else has executed */

  window.addEventListener('load', restoreMenu, false);


  /* run again whenever the window is resized */

  window.addEventListener('resize', restoreMenu, false);

})();

</script>
